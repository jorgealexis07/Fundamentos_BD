drop database tienda_bolsas;

CREATE DATABASE tienda_bolsas;

USE tienda_bolsas;

CREATE TABLE bolsa(
  ID_BOLSA	 	VARCHAR(20) NOT NULL,
  MODELO 	 	VARCHAR(120) NOT NULL,
  PRECIO 	 	DECIMAL(9,2) NOT NULL,
  COLOR 	 	VARCHAR(40) NOT NULL,
  MEDIDA 	 	VARCHAR(90) NOT NULL,
  MARCA 	 	VARCHAR(30) NOT NULL,
  EXISTENCIA_TIENDA  	VARCHAR(10) NOT NULL,
  PRIMARY KEY(ID_BOLSA, MODELO, EXISTENCIA_TIENDA)
)ENGINE=Innodb;

CREATE TABLE almacen(
  ID_ALMACEN 	 	VARCHAR(20) NOT NULL,
  ID_BOLSA1 	 	VARCHAR(20) NOT NULL,
  MODELO1 	 	VARCHAR(120) NOT NULL,
  MARCA1 	 	VARCHAR(30) NOT NULL,
  EXISTENCIA_ALMACEN 	VARCHAR(10) NOT NULL,
  EXISTENCIA_TIENDA1 	VARCHAR(10) NOT NULL,
  PRIMARY KEY(ID_ALMACEN,ID_BOLSA1,MODELO1,MARCA1,EXISTENCIA_TIENDA1),
  INDEX almacen_FKIndex1(ID_BOLSA1),
  INDEX almacen_FKIndex2(EXISTENCIA_TIENDA1),
  INDEX almacen_FKIndex3(MODELO1),
  INDEX almacen_FKIndex4(MARCA1),
  FOREIGN KEY(ID_BOLSA1) REFERENCES bolsa(ID_BOLSA),
  FOREIGN KEY(EXISTENCIA_TIENDA1) REFERENCES bolsa(EXISTENCIA_TIENDA),
  FOREIGN KEY(MODELO1) REFERENCES bolsa(MODELO),
  FOREIGN KEY(MARCA1) REFERENCES bolsa(MARCA)
)ENGINE=Innodb;

CREATE TABLE compras(
  ID_COMPRAS  		INTEGER UNSIGNED NOT NULL,
  PRECIO 		DECIMAL(9,2) NOT NULL,
  ID_PROVEEDOR1 	INTEGER UNSIGNED NOT NULL,
  CANTIDAD_COMPRAS 	INT NOT NULL,
  PRIMARY KEY(ID_COMPRAS),
  INDEX compras_FKIndex1(ID_PROVEEDOR1),
  FOREIGN KEY(ID_PROVEEDOR1) REFERENCES proveedores(ID_PROVEEDOR)
  ON DELETE SET NULL
)ENGINE=Innodb;

CREATE TABLE proveedores(
  ID_PROVEEDOR 		INTEGER UNSIGNED NOT NULL,
  NOMBRE 		VARCAHR(30) NOT NULL,
  PRECIO_PROV 		DECIMAL(9,2) NOT NULL,
  CANTIDAD_PEDIDO 	INT NOT NULL,
  PRIMARY KEY(ID_PROVEEDOR)
)ENGINE=Innodb;

CREATE TABLE vendedor(
  ID_VENDEDOR 		INTEGER UNSIGNED NOT NULL,
  NOMBRE 		VARCHAR(40) NOT NULL,
  TELEFONO  		VARCHAR(40) NOT NULL,
  DOMICILIO 		VARCHAR(40) NOT NULL,
  SUELDO 		DECIMAL(9,2) NOT NULL,
  COMISION 		DECIMAL(9,2) NOT NULL,
  HORARIO 		TIME NOT NULL,
  PRIMARY KEY(ID_VENDEDOR)
)ENGINE=Innodb;

CREATE TABLE venta(
  ID_VENTA 		INTEGER UNSIGNED NOT NULL,
  ID_VENDEDOR1 		INTEGER UNSIGNED NOT NULL,
  ID_COMPRAS1 		INTEGER UNSIGNED NOT NULL,
  PRECIO 		DECIMAL(9,2) NOT NULL,
  PRIMARY KEY(ID_VENTA),
  INDEX venta_FKIndex1(ID_VENDEDOR1),
  FOREIGN KEY(ID_VENDEDOR1) REFERENCES vendedor(ID_VENDEDOR),
  INDEX venta_FKIndex2(ID_COMPRAS1),
  FOREIGN KEY(ID_COMPRAS1) REFERENCES compras(ID_COMPRAS)
  ON DELETE SET NULL
)ENGINE=Innodb;


CREATE TABLE horario(
  ID_HORARIO 		INTEGER UNSIGNED NOT NULL,
  ID_VENDEDOR2 		INTEGER UNSIGNED NOT NULL,
  NOMBRE_VEN 		VARCHAR(40) NOT NULL,
  HORA_ENTRADA 		TIME NOT NULL,
  HORARIO_SALIDA 	TIME NOT NULL,
  RETARDO 		VARCHAR(40) NOT NULL,
  PRIMARY KEY(ID_HORARIO),
  INDEX venta_FKIndex1(ID_VENDEDOR2),
  FOREIGN KEY(ID_VENDEDOR2) REFERENCES vendedor(ID_VENDEDOR),
  INDEX venta_FKIndex2(NOMBRE_VEN),
  FOREIGN KEY(NOMBRE_VEN) REFERENCES vendedor(NOMBRE)
)ENGINE=Innodb;
